4. ADD hooks/useMenu.ts

import { useEffect, useState, useCallback } from "react";

import { fetchMenu } from "../services/menu.service";

import { MenuNode } from "../types/menu.types";

export function useMenu() {
  const [menu, setMenu] = useState<MenuNode[]>([]);

  const [loading, setLoading] = useState(true);

  const [error, setError] = useState("");

  const loadMenu = useCallback(async () => {
    try {
      setLoading(true);

      const data = await fetchMenu();

      setMenu(data as MenuNode[]);
    } catch {
      setError("Failed to load menu");
    } finally {
      setLoading(false);
    }
  }, []);

  useEffect(() => {
    loadMenu();
  }, [loadMenu]);

  return {
    menu,
    loading,
    error,
  };
}


5. ADD RecursiveMenuItem.tsx 


import React, { memo } from "react";

import Menu from "@mui/material/Menu";
import MenuItem from "@mui/material/MenuItem";

import { useNavigate } from "react-router-dom";

import { MenuNode } from "../../types/menu.types";

type Props = {
  item: MenuNode;
};

function RecursiveMenuItemComponent({
  item,
}: Props) {
  const navigate = useNavigate();

  const isDashboard = !!item.dashboardId;

  if (isDashboard) {
    return (
      <MenuItem
        onClick={() =>
          navigate(`/dashboard/${item.dashboardId}`)
        }
      >
        {item.name}
      </MenuItem>
    );
  }

  return (
    <>
      <MenuItem>{item.name}</MenuItem>

      {item.children?.map((child) => (
        <RecursiveMenuItem
          key={child.id}
          item={child}
        />
      ))}
    </>
  );
}

export const RecursiveMenuItem = memo(
  RecursiveMenuItemComponent
);
