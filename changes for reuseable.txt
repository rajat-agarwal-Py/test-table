import {
  Box,
  Button,
  Typography,
} from "@mui/material";

import { useState } from "react";

interface Props {
  text: string;
}

const ExpandableDescription = ({
  text,
}: Props) => {
  const [expanded, setExpanded] =
    useState(false);

  return (
    <Box sx={{ py: 0.5 }}>
      <Typography
        variant="body2"
        sx={{
          color: "#4B5563",

          overflow: "hidden",

          display: "-webkit-box",

          WebkitBoxOrient: "vertical",

          WebkitLineClamp: expanded
            ? "unset"
            : 2,

          lineHeight: 1.4,

          wordBreak: "break-word",
        }}
      >
        {text}
      </Typography>

      {text.length > 100 && (
        <Button
          size="small"
          onClick={() =>
            setExpanded(!expanded)
          }
          sx={{
            p: 0,

            mt: 0.2,

            minWidth: "unset",

            textTransform: "none",

            fontSize: "12px",

            fontWeight: 600,
          }}
        >
          {expanded
            ? "See Less"
            : "See More"}
        </Button>
      )}
    </Box>
  );
};

export default ExpandableDescription;

6. src/pages/datacatalog/DataCatalogPage.tsx

  import {
  Alert,
  Box,
  Tab,
  Tabs,
  Typography,
} from "@mui/material";

import {
  useCallback,
  useMemo,
  useState,
} from "react";

import { useDebounce } from "use-debounce";

import CatalogFilters from "./components/CatalogFilters";

import DataTable from "../../components/table/DataTable";

import StatsCards from "../../components/stats/StatsCards";

import { sourceColumns } from "./sourceColumns";

import { ingestedColumns } from "./ingestedColumns";

import CatalogSkeleton from "../../components/loaders/CatalogSkeleton";

import { useCatalog } from "../../hooks/useCatalog";

const sourceStats = [
  {
    title: "Total Sources",
    value: "1,248",
  },

  {
    title: "CSV Files",
    value: "856",
  },

  {
    title: "Excel Files",
    value: "214",
  },

  {
    title: "BigQuery Tables",
    value: "178",
  },
];

const ingestionStats = [
  {
    title: "Successful Loads",
    value: "98%",
  },

  {
    title: "Failed Loads",
    value: "2%",
  },

  {
    title: "Daily Pipelines",
    value: "42",
  },

  {
    title: "Fresh Datasets",
    value: "1,100",
  },
];

const DataCatalogPage = () => {
  const [search, setSearch] =
    useState("");

  const [activeTab, setActiveTab] =
    useState(0);

  const [debouncedSearch] =
    useDebounce(search, 500);

  const {
    rows,
    isLoading,
    error,
  } = useCatalog();

  const filteredRows = useMemo(() => {
    return rows.filter((item) => {
      return (
        item.fileName
          .toLowerCase()
          .includes(
            debouncedSearch.toLowerCase()
          ) ||

        item.description
          .toLowerCase()
          .includes(
            debouncedSearch.toLowerCase()
          )
      );
    });
  }, [rows, debouncedSearch]);

  const handleSearch =
    useCallback((value: string) => {
      setSearch(value);
    }, []);

  if (isLoading) {
    return <CatalogSkeleton />;
  }

  if (error) {
    return (
      <Alert severity="error">
        Failed to load catalog
      </Alert>
    );
  }

  return (
    <Box sx={{ p: 3 }}>
      <Typography
        variant="h4"
        fontWeight={700}
        mb={1}
      >
        Data Catalog
      </Typography>

      <Typography
        variant="body1"
        color="text.secondary"
        mb={3}
      >
        Discover and explore datasets
      </Typography>

      <Tabs
        value={activeTab}
        onChange={(_, value) =>
          setActiveTab(value)
        }
        sx={{
          mb: 3,

          "& .MuiTabs-indicator": {
            backgroundColor: "#7C3AED",

            height: 3,

            borderRadius: "999px",
          },
        }}
      >
        <Tab
          label="Sources"
          sx={{
            textTransform: "none",

            fontWeight: 600,
          }}
        />

        <Tab
          label="Ingested"
          sx={{
            textTransform: "none",

            fontWeight: 600,
          }}
        />
      </Tabs>

      {activeTab === 0 && (
        <>
          <StatsCards
            stats={sourceStats}
          />

          <CatalogFilters
            search={search}
            setSearch={handleSearch}
          />

          <DataTable
            rows={filteredRows}
            columns={sourceColumns}
            loading={isLoading}
          />
        </>
      )}

      {activeTab === 1 && (
        <>
          <StatsCards
            stats={ingestionStats}
          />

          <DataTable
            rows={[]}
            columns={ingestedColumns}
          />
        </>
      )}
    </Box>
  );
};

export default DataCatalogPage;
